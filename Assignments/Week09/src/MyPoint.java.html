1 /**
  2  * Point class for Assignment 9.1
  3  */
  4 public class MyPoint {
  5     
  6     private int x;
  7     private int y;
  8     
  9     /**
 10      * Default constructor (0,0)
 11      */
 12     public MyPoint() {
 13         x = 0;
 14         y = 0;
 15     }
 16     
 17     /**
 18      * Constructor
 19      * @param x
 20      * @param y
 21      */
 22     public MyPoint(int x, int y) {
 23         this.x = x;
 24         this.y = y;
 25     }
 26 
 27 
 28     /**
 29      * @return the x coordinate
 30      */
 31     public int getX() {
 32         return x;
 33     }
 34 
 35     /**
 36      * Set the x coordinate
 37      * @param x
 38      */
 39     public void setX(int x) {
 40         this.x = x;
 41     }
 42 
 43 
 44     /**
 45      * @return the y coordinate
 46      */
 47     public int getY() {
 48         return y;
 49     }
 50 
 51     /**
 52      * Set the y coordinate
 53      * @param y
 54      */
 55     public void setY(int y) {
 56         this.y = y;
 57     }
 58     
 59     /**
 60      * Calculate the distance between this point and another point.
 61      * @param p
 62      * @return
 63      */
 64     public double distance(MyPoint p) {
 65         return distance(p.getX(), p.getY());
 66     }
 67     
 68     
 69     /**
 70      * Calculate the distance between this point and another point
 71      * having the given x, y coordinates.
 72      * @param xCoord
 73      * @param yCoord
 74      * @return
 75      */
 76     public double distance(int xCoord, int yCoord) {
 77         // Pythagorean Theorem: a^2 + b^2 = c^2
 78         //                      a*a + b*b = c*c
 79         //                  c = sqrt(a*a + b*b)
 80 
 81         // Find the distance between the x coordinates.
 82         double a = x - xCoord;
 83 
 84         // Find the distance between the y coordinates.
 85         double b = y - yCoord;
 86 
 87         // calculate and return the length of the hypotenuse
 88         return Math.sqrt((a*a) + (b*b));
 89     }
 90     
 91     
 92     /**
 93      * Calculate the distance between the two points.
 94      * @param point1
 95      * @param point2
 96      * @return
 97      * @throws IllegalArgumentException if either point is null
 98      */
 99     public static double distance(MyPoint point1, MyPoint point2) {
100         if (point1 == null || point2 == null) {
101             throw new IllegalArgumentException("Points must not be null");
102         }
103         return point1.distance(point2);
104     }
105 
106     /**
107      * Return a string representation of the point.
108      * @return
109      */
110     @Override
111     public String toString() {
112         return String.format("(%d,%d)", x, y);
113     }
114 
115 }
