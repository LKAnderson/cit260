1 import java.util.Scanner;
 2 
 3 /**
 4  * Convert a hex character to its binary representation.
 5  */
 6 public class W4dot1_Bitwise {
 7 
 8     public static void main(final String[] argv) {
 9 
10         try (Scanner scanner = new Scanner(System.in)) {
11 
12             // prompt the user for the hex digit.
13             System.out.print("Enter a hex digit: ");
14             String hexString = scanner.nextLine().trim();
15             hexString = hexString.toUpperCase();
16 
17             // Make sure they only entered a single digit.
18             if (hexString.length() != 1) {
19                 if (hexString.length() == 0) {
20                     hexString = "<empty>";
21                 }
22 
23                 System.err.println(hexString + " is an invalid input");
24                 System.exit(1);
25             }
26 
27             // Get the first (only) character
28             final char hexDigit = hexString.charAt(0);
29 
30             // default to an empty string
31             String binaryString = "";
32 
33             // Find the binary string according to the relative offset
34             // of the character from its root character.
35             if (hexDigit >= 'A' && hexDigit <= 'F') {
36                 binaryString = getBinaryString(10 + (hexDigit - 'A'));
37 
38             } else if (hexDigit >= '0' && hexDigit <= '9') {
39                 binaryString = getBinaryString(hexDigit - '0');
40 
41             } else {
42                 System.err.printf("%c is an invalid input%n", hexDigit);
43                 // Exiting the whole program from an if-else-if ladder
44                 // isn't really the best form, but it will do for this
45                 // lesson.
46                 System.exit(1);
47             }
48 
49             // Print it out.
50             System.out.printf("The binary value is %s%n", binaryString);
51         }
52     }
53 
54     /**
55      * Build a binary string using bitwise shifting and operators
56      * 
57      * @param ch The value to convert to a binary string.
58      */
59     private static String getBinaryString(final int ch) {
60 
61         // Use a StringBuilder so we don't have to waste
62         // memory concatenating strings.
63         StringBuilder binaryString = new StringBuilder();
64 
65         // Loop through bits 0 through 4
66         for (int i = 0; i < 4; i++) {
67 
68             // Shift the bit so that it is in position 0
69             // then AND it with 1. If the bit is 1, 
70             // then 1 & 1 = 1. Otherwise 1 & 0 = 0.
71             binaryString.append((ch >> i) & 1);
72         }
73 
74         // We built it in reverse, so reverse it
75         // to put it in the correct order.
76         return binaryString.reverse().toString();
77     }
78 }
