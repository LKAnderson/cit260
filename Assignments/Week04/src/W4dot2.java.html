1 import java.util.Scanner;
 2 import java.text.DecimalFormat;
 3 
 4 /**
 5  * Produce a payroll report.
 6  */
 7 public class W4dot2 {
 8 
 9     // Tax constants
10     private final static double FEDERAL_TAX_RATE = 0.20;
11     private final static double STATE_TAX_RATE = 0.09;
12 
13     public static void main(String[] argv) {
14 
15         try (Scanner scanner = new Scanner(System.in)) {
16 
17             DecimalFormat format = new DecimalFormat("$#,##0.00");
18 
19             // Prompt for inputs
20             System.out.print("Enter employee's name: ");
21             String employeeName = scanner.nextLine().trim();
22             if (employeeName.length() == 0) {
23                 System.err.println("Invalid name");
24                 return;
25             }
26 
27             System.out.print("Enter number of hours worked in a week: ");
28             double hoursWorked = scanner.nextDouble();
29             if (hoursWorked < 0) {
30                 System.err.println("Invalid input");;
31                 return;
32             }
33 
34             System.out.print("Enter the hourly payrate: ");
35             double payRate = scanner.nextDouble();
36             if (payRate <= 0) {
37                 System.err.println("Invalid input");
38                 return;
39             }
40 
41             // do the calculations
42             double grossPay = hoursWorked * payRate;
43             double federalWitholding = grossPay * FEDERAL_TAX_RATE;
44             double stateWitholding = grossPay * STATE_TAX_RATE;
45             double totalDeductions = federalWitholding + stateWitholding;
46             double netPay = grossPay - totalDeductions;
47 
48             // Now print the report
49             System.out.format("Employee Name:        %12s%n", employeeName);
50             System.out.format("Hours worked:         %12.2f%n", hoursWorked);
51             System.out.format("Pay Rate:             %12s/hr%n", format.format(payRate));
52             System.out.format("Gross Pay:            %12s%n", format.format(grossPay));
53             System.out.format("  Federal Witholding: %12s%n", format.format(federalWitholding));
54             System.out.format("  State Witholding:   %12s%n", format.format(stateWitholding));
55             System.out.format("  Total Deduction:    %12s%n", format.format(totalDeductions));
56             System.out.format("Net Pay:              %12s%n", format.format(netPay));
57         }
58     }
59 }
