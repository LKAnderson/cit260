1 import java.util.Scanner;
 2 
 3 /**
 4  * Convert a hex character to its binary representation.
 5  */
 6 public class W4dot1_IntegerToBinaryString {
 7 
 8     public static void main(String[] argv) {
 9 
10         try (Scanner scanner = new Scanner(System.in)) {
11 
12             // prompt the user for the hex digit.
13             System.out.print("Enter a hex digit: ");
14             String hexString = scanner.nextLine().trim();
15             hexString = hexString.toUpperCase();
16 
17             // Make sure they only entered a single digit.
18             if (hexString.length() != 1) {
19                 if (hexString.length() == 0) {
20                     hexString = "<empty>";
21                 }
22 
23                 System.err.println(hexString + " is an invalid input");
24                 System.exit(1);
25             }
26 
27             // Get the first (only) character
28             char hexDigit = hexString.charAt(0);
29 
30             // default to an empty string
31             String binaryString = "";
32 
33             // Find the binary string according to the relative offset
34             // of the character from its root character.
35             if (hexDigit >= 'A' && hexDigit <= 'F') {
36                 binaryString = Integer.toBinaryString(10 + (hexDigit - 'A'));
37 
38             } else if (hexDigit >= '0' && hexDigit <= '9') {
39                 binaryString = Integer.toBinaryString(hexDigit - '0');
40 
41             } else {
42                 System.err.printf("%c is an invalid input%n", hexDigit);
43                 // Exiting the whole program from an if-else-if ladder
44                 // isn't really the best form, but it will do for this
45                 // lesson.
46                 System.exit(1);
47             }
48 
49             // Make sure the string is 4 digits
50             binaryString = "0".repeat(4 - binaryString.length()) + binaryString;
51 
52             // Print it out.
53             System.out.printf("The binary value is %s%n", binaryString);
54         }
55     }
56 }
