1 import java.util.Scanner;
 2 
 3 /**
 4  * Convert a hex character to its binary representation.
 5  */
 6 public class W4dot1 {
 7 
 8     // Could use a switch statement, but arrays are much faster and can
 9     // be used repeatedly.
10     private final static String[] BINARY_DIGITS = new String[] { 
11             "0000", // 0
12             "0001", // 1
13             "0010", // 2
14             "0011", // 3
15             "0100", // 4
16             "0101", // 5
17             "0110", // 6
18             "0111", // 7
19             "1000", // 8
20             "1001", // 9
21             "1010", // A (10)
22             "1011", // B (11)
23             "1100", // C (12)
24             "1101", // D (13)
25             "1110", // E (14)
26             "1111", // F (15)
27     };
28 
29     public static void main(String[] argv) {
30 
31         try (Scanner scanner = new Scanner(System.in)) {
32 
33             // prompt the user for the hex digit.
34             System.out.print("Enter a hex digit: ");
35             String hexString = scanner.nextLine().trim();
36             hexString = hexString.toUpperCase();
37 
38             // Make sure they only entered a single digit.
39             if (hexString.length() != 1) {
40                 if (hexString.length() == 0) {
41                     hexString = "<empty>";
42                 }
43 
44                 System.err.println(hexString + " is an invalid input");
45                 System.exit(1);
46             }
47 
48             // Get the first (only) character
49             char hexDigit = hexString.charAt(0);
50 
51             // default to an empty string
52             String binaryString = "";
53 
54             // Find the binary string according to the relative offset
55             // of the character from its root character.
56             if (hexDigit >= 'A' && hexDigit <= 'F') {
57                 binaryString = BINARY_DIGITS[10 + (hexDigit - 'A')];
58 
59             } else if (hexDigit >= '0' && hexDigit <= '9') {
60                 binaryString = BINARY_DIGITS[hexDigit - '0'];
61 
62             } else {
63                 System.err.printf("%c is an invalid input%n", hexDigit);
64                 // Exiting the whole program from an if-else-if ladder
65                 // isn't really the best form, but it will do for this
66                 // lesson.
67                 System.exit(1);
68             }
69 
70             // Print it out.
71             System.out.printf("The binary value is %s%n", binaryString);
72         }
73     }
74 }
