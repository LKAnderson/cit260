1 /**
 2  * Temperature converter.
 3  */
 4 public class W6dot2 {
 5 
 6     /**
 7      * Convert C to F
 8      * @param celsius the degrees in celsius
 9      * @return the equivalent degrees in fahrenheit
10      */
11     public static double celsiusToFahrenheit(double celsius) {
12         return (9.0/5 * celsius) + 32;
13     }
14 
15     /**
16      * Convert F to C
17      * @param fahrenheit the degrees in fahrenheit
18      * @return the equivalent degrees in celsius
19      */
20     public static double fahrenheitToCelsius(double fahrenheit) {
21         return (fahrenheit - 32) / (9.0/5);
22     }
23 
24 
25     public static void main(String[] argv) {
26 
27         String tableHeader = "Celsius   Fahrenheit   |   Fahrenheit   Celsius";
28         String tableLine =   "-----------------------------------------------";
29         
30         String tableTemplate = "%7.1f   %10.1f   |   %10.1f   %7.2f\n";
31 
32         System.out.println(tableHeader);
33         System.out.println(tableLine);
34 
35         double celsius = 40.0;
36         double fahrenheit = 120.0;
37 
38         for (int i=0; i < 10; i++, celsius -= 1.0, fahrenheit -= 10.0 ){
39             double c2f = celsiusToFahrenheit(celsius);
40             double f2c = fahrenheitToCelsius(fahrenheit);
41 
42             System.out.printf(tableTemplate, celsius, c2f, fahrenheit, f2c);
43         }
44         
45     }
46 
47 }
