 2  * Future Value Calculator.
 3  */
 4 
 5 import java.text.DecimalFormat;
 6 import java.util.Scanner;
 7 
 8 public class W6dot1 {
 9 
10     /**
11      * Calculate the future value of an investment.
12      * @param investmentAmount
13      * @param monthlyInterestRate
14      * @param years
15      * @return
16      */
17     public static double futureInvestmentValue(
18                             double investmentAmount,
19                             double monthlyInterestRate,
20                             int years ) {
21         
22         return investmentAmount * Math.pow(1 + monthlyInterestRate, years*12);
23     }
24 
25     public static void main(String[] argv) {
26 
27         DecimalFormat formatter = new DecimalFormat("$#,##0.00");
28 
29         try (Scanner scanner = new Scanner(System.in)) {
30             System.out.print("The amount invested: ");
31             double amountInvested = scanner.nextDouble();
32             if (amountInvested <= 0) {
33                 System.err.println("Amount must be greater than zero.");
34                 System.exit(1);
35             }
36 
37             System.out.print("Annual interest rate: ");
38             double yearlyInterest = scanner.nextDouble();
39             if (yearlyInterest < 0 || yearlyInterest >= 100) {
40                 System.err.println("Invalid interest rate");
41                 System.exit(1);
42             }
43 
44             yearlyInterest = yearlyInterest / 100.0;
45 
46             String tableHeader   = "Years     Future Value\n";
47             String tableTemplate = "%-5d     %s\n";
48 
49             System.out.print(tableHeader);
50 
51             double futureValue = 0;
52             for (int year = 1; year <= 10; year++) {
53                 futureValue = futureInvestmentValue(amountInvested, yearlyInterest/12, year);
54                 System.out.printf(tableTemplate, year, formatter.format(futureValue));
55             }
56 
57         }
58 
59     }
60     
61 }
