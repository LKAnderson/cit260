1 import java.util.Scanner;
 2 
 3 /**
 4  * Provide methods for getting input from users.
 5  */
 6 public class Inputs {
 7 
 8     /**
 9      * Read double values from the user.
10      * @param count The number of values to read
11      * @param prompt The message to print
12      * @return
13      */
14     public static double[] getDoubles(final int count, final String prompt) {
15 
16         double[] inputs = new double[count];
17 
18         try (Scanner scanner = new Scanner(System.in)) {
19             System.out.println(prompt);
20 
21             int stringCount = 0;
22             while (stringCount < count) {
23                 System.out.format("%d/%d: ", stringCount + 1, count);
24                 inputs[stringCount] = scanner.nextDouble();
25                 stringCount += 1;
26             }
27         }
28 
29         return inputs;
30     }
31 
32     /**
33      * Convert an array of strings to an array of doubles.
34      * @param input the array of strings
35      * @return the array of doubles.
36      */
37     public static double[] getDoubles(String[] input) {
38         double[] output = new double[input.length];
39         for (int i=0; i < input.length; i++) {
40             try {
41                 output[i] = Double.parseDouble(input[i]);
42             } catch (NumberFormatException exception) {
43                 // We'll set it to 0. We'll learn how to 
44                 // handle this error better in Week 11.
45                 output[i] = 0;
46             }
47         }
48         return output;
49     }
