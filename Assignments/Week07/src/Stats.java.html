1 /**
 2  * Utility class to provide Statistical calculations.
 3  */
 4 public class Stats {
 5 
 6     /**
 7      * Calculate the mean of an array of double values.
 8      * @param values
 9      * @return
10      */
11     public static double mean(final double[] values) {
12         double total = 0;
13         for (double value: values) {
14             total += value;
15         }
16         return total/values.length;
17     }
18 
19     /**
20      * Find the average of an array of double values (This
21      * method is the same as mean().
22      * @param values
23      * @return
24      */
25     public static double average(final double[] values){
26         return mean(values);
27     }
28 
29     /**
30      * Calculate the standard deviation of an array of double values.
31      * @param values
32      * @return
33      */
34     public static double deviation(final double[] values) {
35         double mean = mean(values);
36 
37         double sumDeviation = 0;
38         for (double num: values) {
39             sumDeviation += Math.pow(num - mean, 2);
40         }
41 
42         return Math.sqrt(sumDeviation / (values.length-1));
43     }
44 }
