1 import java.util.Scanner;
 2 
 3 /**
 4  * Convert celsius to fahrenheight
 5  */
 6 public class W2dot1 {
 7     public static void main(String[] argv) {
 8 
 9         // Open a scanner using the keyboard as input. Close it when
10         // the program ends.
11         try (var scanner = new Scanner(System.in)) {
12 
13             // Description
14             System.out.println("This program converts celsius to fahrenheit.\n");
15 
16             // Prompt for input

19

22 
23             // Print report

28     }
