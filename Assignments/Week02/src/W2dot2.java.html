1 import java.util.Scanner;
 2 
 3 /**
 4  * Create a tax+tip calculator. Print all of the details of the transaction.
 5  */
 6 public class W2dot2 {
 7 
 8     /**
 9      * The tax rate to be applied to the transaction.
10      */
11     private static final double TAX_RATE = 0.032;
12 
13     public static void main(String[] argv) {
14 
15         // Open a scanner using the keyboard as input. Close it automatically
16         // at the end of the program.
17         try (var scanner = new Scanner(System.in)) {
18 
19             // Description
20             System.out.println("This is a tax + tip calculator.\n");
21 
22             // Prompt and gather inputs.
23             System.out.print("Enter the meal cost: ");
24             var cost = scanner.nextDouble();
25 
26             System.out.print("Enter the tip percentage: ");
27             var tipRate = scanner.nextDouble() * 0.01;
28 
29             // calculate the tip and tax.
30             var tip = cost * tipRate;
31             var tax = cost * TAX_RATE;
32 
33             // total it all
34             var total = cost + tip + tax;
35 
36             // Print the report
37             System.out.format("The tip is $%.02f\n", tip);
38             System.out.format("The tax is $%.02f\n", tax);
39             System.out.format("The total is $%.02f\n", total);
40             System.out.println("Good-bye");
41         }
42     }
43 }
