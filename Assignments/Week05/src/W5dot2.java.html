1 /**
 2  * Print all numbers between 100 and 1000
 3  * that are evenly divisible by both 5 and 6.
 4  */
 5 public class W5dot2 {
 6 
 7     private static final int MAX_PER_LINE = 10;
 8 
 9     public static void main(String[] argv) {
10 
11         // Loop from 100 to 1000, by one
12         // also initialize a print counter to 0
13         int printCounter = 0;
14 
15         for (int i=100; i <= 1000; i++) {
16 
17             // if the loop counter % 5 is 0 AND the loop counter % 6 is 0
18             // then print out the number
19             if ( (i % 5 == 0) && (i % 6 == 0)) {
20 
21                 System.out.printf("%-4d", i);
22 
23                 // increment the print counter and check if it's 10. If so, print a
24                 // newline, the reset it for the next line
25                 printCounter += 1;
26 
27                 if (printCounter % MAX_PER_LINE == 0) {
28                     System.out.println();
29                 }
30             }
31         }
32     }
33 }
