1 /**
 2  * A Rectangle class to model a rectangle in code.
 3  * @author kent
 4  */
 5 public class Rectangle {
 6     
 7     private double width;
 8     private double height;
 9     
10     /**
11      * Default constructor. This is not too useful since we're
12      * not providing a way to update the properties once the 
13      * object is created.
14      */
15     public Rectangle() {
16         width = 1;
17         height = 1;
18     }
19     
20     /**
21      * Constructor. Create a new Rectangle object with the given
22      * width and height.
23      * @param width
24      * @param height
25      */
26     public Rectangle(double width, double height) {
27         this.width = width;
28         this.height = height;
29     }
30     
31     public double getWidth() {
32         return width;
33     }
34     
35     public void setWidth(double width) {
36         this.width = width;
37     }
38 
39     public double getHeight() {
40         return height;
41     }
42 
43     public void setHeight(double height) {
44         this.height = height;
45     }
46     
47     /**
48      * Calculate the area of the rectangle.
49      * @return
50      */
51     public double getArea() {
52         return width * height;
53     }
54     
55     /**
56      * Calculate the perimeter of the rectangle.
57      * @return
58      */
59     public double getPerimeter() {
60         return (2 * width) + (2 * height);
61     }
62 
63 }
