1 import java.io.File;
 2 import java.io.IOException;
 3 import java.io.PrintWriter;
 4 import java.util.ArrayList;
 5 import java.util.Scanner;
 6 
 7 /**
 8  * This class provides methods to store and load
 9  * data in files.
10  */
11 public class Storage {
12 
13     /**
14      * Stores the array of data in the file specified by the filepath.
15      * @param filepath The path to the file
16      * @param data The data to be stored.
17      * @throws IOException
18      */
19     public static void storeData(String filepath, int[] data) throws IOException {
20         try (var output = new PrintWriter(filepath)) {
21             for (int d : data) {
22                 output.println(d);
23             }
24         }
25     }
26 
27     /**
28      * Loads the array of data from the file specified by filepath
29      * @param filepath The path to the file.
30      * @return
31      * @throws IOException
32      */
33     public static int[] loadData(String filepath) throws IOException  {
34         // Check if file exists.
35         var file = new File(filepath);
36         if (file.exists() == false) {
37             throw new IOException("File not found: " + filepath);
38         }
39 
40         if (file.canRead() == false) {
41             throw new IOException("File cannot be read: " + filepath);
42         }
43 
44         var data = new ArrayList<Integer>();
45 
46         try (var input = new Scanner(new File(filepath))) {
47             while (input.hasNextLine()) {
48                 String line = input.nextLine().trim();
49                 data.add(Integer.parseInt(line));
50             }
51         } catch (NumberFormatException ex) {
52             throw new IOException("Invalid number value in file", ex);
53         }
54 
55         return Utils.convertListToArray(data);
56     }
57 }
