 3 /**

 8     /**

10      * @param data
11      * @return
12      * @throws IllegalArgumentException if data is null
13      */
14     public static int[] convertListToArray(ArrayList<Integer> data) {
15         if (data == null) {
16             throw new IllegalArgumentException("Data must not be null");
17         }
18 
19         int[] array = new int[data.size()];
20         for (int i=0; i < data.size(); i++) {
21             array[i] = data.get(i);
22         }
23 
24         return array;
25     }
26 
27     /**
28      * Convert ArrayList of Integer to Integer[]
29      * @param data
30      * @return
31      * @throws IllegalArgumentException if data is null
32      */
33     public static Integer[] convertListToIntegerArray(ArrayList<Integer> data) {
34         if (data == null) {
35             throw new IllegalArgumentException("Data must not be null");
36         }
37         Integer[] arrayData = new Integer[data.size()];
