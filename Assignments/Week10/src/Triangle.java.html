1 /**
  2  * Implementation of the Triangle class
  3  */
  4 public class Triangle extends GeometricObject {
  5     private double side1;
  6     private double side2;
  7     private double side3;
  8 
  9     /**
 10      * Default constructor
 11      */
 12     public Triangle() {
 13         super();
 14         side1 = 1;
 15         side2 = 1;
 16         side3 = 1;
 17     }
 18 
 19     /**
 20      * Full Constructor
 21      * @param side1
 22      * @param side2
 23      * @param side3
 24      * @param color
 25      * @param filled
 26      */
 27     public Triangle(double side1, double side2, double side3,
 28                     String color, boolean filled) {
 29         super(color, filled);
 30         this.side1 = side1;
 31         this.side2 = side2;
 32         this.side3 = side3;
 33     }
 34 
 35     /**
 36      * Get side1
 37      * @return
 38      */
 39     public double getSide1() {
 40         return side1;
 41     }
 42 
 43     /**
 44      * Updated side1
 45      * @param side1
 46      */
 47     public void setSide1(double side1) {
 48         this.side1 = side1;
 49     }
 50 
 51     /**
 52      * Get side2
 53      * @return
 54      */
 55     public double getSide2() {
 56         return side2;
 57     }
 58 
 59     /**
 60      * Updated side2
 61      * @param side2
 62      */
 63     public void setSide2(double side2) {
 64         this.side2 = side2;
 65     }
 66 
 67     /**
 68      * Get side3
 69      * @return
 70      */
 71     public double getSide3() {
 72         return side3;
 73     }
 74 
 75     /**
 76      * Updated side3
 77      * @param side3
 78      */
 79     public void setSide3(double side3) {
 80         this.side3 = side3;
 81     }
 82 
 83     /**
 84      * Calculate the area of the triangle. Since we only know the
 85      * three sides, we must use Heron's formula. You can see
 86      * an example of it at https://www.mathopenref.com/heronsformula.html.
 87      */
 88     @Override
 89     public double getArea() {
 90         double p = (side1 + side2 + side3) / 2.0;
 91         double area = Math.sqrt(p * (p-side1) * (p-side2) * (p-side3));
 92         return area;
 93     }
 94 
 95     /**
 96      * return a string representation of our rectangle
 97      */
 98     @Override
 99     public String toString() {
100         return String.format("Side1: %.2f, Side2: %.2f, Side3: %.2f\n%s",
101                 side1, side2, side3, super.toString());
102     }
103 }
