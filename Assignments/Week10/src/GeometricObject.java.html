1 import java.util.Date;
 2 
 3 /**
 4  * Base class for our 2D geometric objects
 5  */
 6 public class GeometricObject {
 7     protected String color = "white";
 8     protected boolean filled;
 9     protected Date dateCreated;
10 
11     /**
12      * Construct a default geometric object.
13      */
14     public GeometricObject() {
15         dateCreated = new Date();
16     }
17 
18     /**
19      * Construct a geometric object with the specified color and filled value.
20      */
21     public GeometricObject(String color, boolean filled) {
22         dateCreated = new Date();
23         this.color = color;
24         this.filled = filled;
25     }
26 
27     /**
28      * Return color
29      */
30     public String getColor() {
31         return color;
32     }
33 
34     /**
35      * Set a new color.
36      */
37     public void setColor(String color) {
38         this.color = color;
39     }
40 
41     /**
42      * Return filled
43      */
44     public boolean isFilled() {
45         return filled;
46     }
47 
48     /**
49      * Set a new filled value
50      */
51     public void setFilled(boolean filled) {
52         this.filled = filled;
53     }
54 
55     /**
56      * Get date created
57      */
58     public Date getDateCreated() {
59         return dateCreated;
60     }
61 
62     /**
63      * Calculate the area
64      */
65     public double getArea() {
66         return 0;
67     }
68 
69     /**
70      * Return string representation of this object.
71      */
72     @Override
73     public String toString() {
74         return String.format("Created on %s\ncolor: %s and filled: %b",
75                 dateCreated.toString(), color, filled);
76     }
77 }
