40         System.out.format("Entering Triangle #%d...%n", shapeNumber);
41         System.out.print("Please enter a color: ");

43         scanner.reset();
44 
45         String color = scanner.nextLine().trim();

47             System.err.println("Please enter a color");

49         }
50 
51         System.out.print("Is the rectangle filled (true/false)? ");
52         boolean filled = scanner.nextBoolean();

54         System.out.print("Enter the lengths of the three sides: ");
55         double side1 = scanner.nextDouble();

57         double side3 = scanner.nextDouble();
