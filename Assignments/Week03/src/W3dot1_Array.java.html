1 import java.util.Scanner;
 2 
 3 /**
 4  * Implement Programming Exercise 3.5 from the Chapter.
 5  *
 6  * (Find future dates) Write a program that prompts the user to enter
 7  * an integer for today's day of the week (Sunday is 0, Monday is 1, ...,
 8  * and Saturday is 6). Also prompt the user to enter the number of days
 9  * after today for a future day and display the future day of the week.
10  *
11  * Here is a sample run:
12  *
13  * ```
14  * Enter today's day: 1
15  * Enter the number of days elapsed since today: 3
16  * Today is Monday and the future day is Thursday
17  * ```
18  *
19  * @author kent
20  *
21  */
22 public class W3dot1_Array {
23 
24     public static void main(String[] argv) {
25 
26         // Sometimes an array lookup is better than a switch statment.
27         // But not always. In this case, the cases of the switch would
28         // would be consecutive integers, perfect for indexing into
29         // an array.
30         String[] weekdays = new String[] {
31                 "Sunday",
32                 "Monday",
33                 "Tuesday",
34                 "Wednesday",
35                 "Thursday",
36                 "Friday",
37                 "Saturday"
38         };
39 
40         try (Scanner scanner = new Scanner(System.in)){
41 
42             System.out.println(
43                     "Given today's day of the week and some number of days in the future\n"
44                             + "this program will tell you the day of the week for the future day.");
45 
46             // prompt for inputs, store in variables
47             System.out.print("Enter today's day of the week (0 for Sunday, 1 for Monday, etc.): ");
48             int today = scanner.nextInt();
49 
50             if (today < 0 || today > 6) {
51                 System.err.println("Invalid weekday. Please enter a number between 0 and 6.");
52                 System.exit(1);
53             }
54 
55             System.out.print("Enter the number of days in the future: ");
56             int future = scanner.nextInt();
57 
58             if (future < 0) {
59                 System.err.println("Invalid future day. You must enter a positive integer.");
60                 System.exit(2);
61             }
62 
63             // calculate weekday names
64             // This line guarantees that invalid data won't mess us up.
65             // 7 % 7 = 0, which is Sunday again.
66             String weekdayToday = weekdays[today % 7];
67 
68             String weekdayFuture = weekdays[(today + future) % 7];
69 
70             // print them out
71             System.out.printf("Today is %s and the future day is %s", weekdayToday, weekdayFuture);
72             System.out.println();
73             System.out.println("Good-bye.");
74         }
75 
76     }
77 
78 
79 }
