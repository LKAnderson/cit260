 3 /**
 4  * Implement Programming Exercise 3.5 from the Chapter.
 5  *
 6  * (Find future dates) Write a program that prompts the user to enter
 7  * an integer for today's day of the week (Sunday is 0, Monday is 1, ...,
 8  * and Saturday is 6). Also prompt the user to enter the number of days
 9  * after today for a future day and display the future day of the week.
10  *
11  * Here is a sample run:
12  *
13  * ```
14  * Enter today's day: 1
15  * Enter the number of days elapsed since today: 3
16  * Today is Monday and the future day is Thursday
17  * ```
18  *
19  * @author kent
20  *
21  */
22 public class W3dot1_Reusable {
23 
24     public static void main(String[] argv) {
25 
26         try (Scanner scanner = new Scanner(System.in)){
27 
28             System.out.println(
29                     "Given today's day of the week and some number of days in the future\n"
30                             + "this program will tell you the day of the week for the future day.");
31 
32             // Prompt for inputs and store in variables
33             System.out.print("Enter today's day of the week (0 for Sunday, 1 for Monday, etc.): ");
34             int today = scanner.nextInt();
35 
36             if (today < 0 || today > 6) {
37                 System.err.println("Invalid weekday. Please enter a number between 0 and 6.");
38                 System.exit(1);
39             }
40 
41             System.out.print("Enter the number of days in the future: ");
42             int future = scanner.nextInt();
43 
44             if (future < 0) {
45                 System.err.println("Invalid future day. You must enter a positive integer.");
46                 System.exit(2);
47             }
48 
49             // Calculate weekday names
50             String weekdayToday = getWeekdayName(today);
51             String weekdayFuture = getWeekdayName((today + future) % 7);
52 
53             // Print them out
54             System.out.printf("Today is %s and the future day is %s", weekdayToday, weekdayFuture);
55 
56             System.out.println();
57             System.out.println("Good-bye.");
58         }
59     }
60 
61     /**
62      * Return the name of the week day
63      * @param weekday The day of the week (0 = Sunday, 6 = Saturday)
64      * @return
65      */
66     private static String getWeekdayName(int weekday) {
67         String name = null;
68 
69         switch(weekday) {
70             case 0:
71                 name = "Sunday";
72                 break;
73             case 1:
74                 name = "Monday";
75                 break;
76             case 2:
77                 name = "Tuesday";
78                 break;
79             case 3:
80                 name = "Wednesday";
81                 break;
82             case 4:
83                 name = "Thursday";
84                 break;
85             case 5:
86                 name = "Friday";
87                 break;
88             case 6:
89                 name = "Saturday";
90                 break;
91         }
92 
93         return name;
94     }
