1 import java.util.Scanner;
 2 
 3 /**
 4  * (Find the number of days in a month) Write a program that prompts
 5  * the user to enter the month and year and displays the number of
 6  * days in the month. For example, if the user entered month 2 and
 7  * year 2012, the program should display that February 2012 has 29
 8  * days. If the user entered 3 and year 2015, the program should display
 9  * that March 2015 has 31 days.
10  */
11 public class W3dot2 {
12 
13     public static void main(String[] argv) {
14 
15         // Some constants to make things easier
16         // to read down below.
17         final int FEB = 2;
18         final int APR = 4;
19         final int JUN = 6;
20         final int SEP = 9;
21         final int NOV = 11;
22 
23         String[] months = new String[] {
24                 "January", "February", "March",    "April", "May",    "June",
25                 "July",    "August", "September", "October", "November", "Decemer"
26         };
27 
28         try (Scanner scanner = new Scanner(System.in)) {
29 
30             System.out.println(
31                 "Given a year and a month in that year, this program will tell you\n"
32                 + "the number of days in that month.");
33             System.out.println();
34 
35             // prompt for inputs, store in variables
36 
37             System.out.print("Enter a year: ");
38             int year = scanner.nextInt();
39 
40             System.out.print("Enter a value for the month (1=Jan, 2=Feb, etc.): ");
41             int month = scanner.nextInt();
42 
43 
44             if (! (month >= 1 && month <= 12)) {
45                 System.err.println("Invalid month");
46                 System.exit(1);
47             }
48 
49             if (year < 0) {
50                 System.err.println("Year must be positive");
51                 System.exit(2);;
52             }
53 
54             // Calculate days in the month
55 
56             // See if it's leap year. We'll use this in the switch, down below
57             boolean isLeapYear = (year % 4 == 0 && year % 100 != 0) 
58                 || year % 400 == 0;
59             
60             // The default case is 31 days
61             int days = 31;
62 
63             // Check for non-default cases
64             switch(month) {
65                 // first the 30-day months
66                 case APR:
67                 case JUN:
68                 case SEP:
69                 case NOV:
70                     days = 30;
71                     break;
72                 // finally, handle poor little February
73                 case FEB:
74                     days = isLeapYear == true ? 29 : 28;
75                     break;
76             }
77 
78             // Print out the report.
79             System.out.printf("%s %d has %d days\n", months[month-1], year, days);
80             System.out.println();
81             System.out.println("Good-bye");
82         }
83     }
84 }
